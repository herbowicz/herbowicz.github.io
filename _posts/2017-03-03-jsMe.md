Basic function to format css-attribute to js style.camelCase

{% highlight javascript %}
function jsMe(css) {
  var pos = css.indexOf("-");
  css = css.slice(0,pos)
       + css.charAt(pos+1).toUpperCase()
       + css.slice(pos+2,css.length);
  return css;
}
console.log(jsMe("font-weight"));  // fontWeight
{% endhighlight %}
